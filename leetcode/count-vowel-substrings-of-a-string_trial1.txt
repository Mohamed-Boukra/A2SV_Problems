class Solution:
    def countVowelSubstrings(self, word: str) -> int:
        vowels = set('aeiou')
        count = 0
        for i in range(len(word)):
            for j in range(i + 1, len(word) + 1):
                sub = word[i:j]
                if set(sub) == vowels:
                    count += 1
        return count
