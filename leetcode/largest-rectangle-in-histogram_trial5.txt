class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        
        stack = []
        result = 0
        n = len(heights)
        for i in range(n):
            while stack and heights[i] < heights[stack[-1]]:
                index = stack.pop()
                left = stack[-1] if stack else -1
                area = heights[index]*(i-left-1)
                result = max(result, area)
            stack.append(i)
        while stack:
            index = stack.pop()
            left = stack[-1] if stack else -1
            area = heights[index]*(n-left-1)
            result = max(result, area)
        return result
