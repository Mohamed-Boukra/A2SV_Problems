class Solution:
    def longestSemiRepetitiveSubstring(self, s: str) -> int:
        l = 0
        maxlen = 0
        notvaliderepcount = 0
        for r in range(len(s)):
            if r > 0 and s[r] == s[r-1]:
                notvaliderepcount += 1
            while notvaliderepcount > 1:
                if s[l] == s[l+1]:
                    notvaliderepcount -=1
                l+=1
            maxlen = max(maxlen , r-l+1)
        return maxlen
