class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        vowels = set('aeiou')
        count = sum(c in vowels for c in s[:k])
        best = count
        for i in range(k , len(s)):
            count += (s[i] in vowels) - (s[i-k] in vowels)
            best = max(best, count)
        return best
