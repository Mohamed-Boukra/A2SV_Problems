class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        def atMost(k):
            left = 0
            odds = 0
            res = 0
            for right in range(len(nums)):
                odds += nums[right] % 2
                while odds > k:
                    odds -= nums[left] % 2
                    left += 1
                res += right - left + 1
            return res
        
        return atMost(k) - atMost(k - 1)
