class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        n = len(nums)
        pre = [1]*(n+1)
        su = [1]*(n+1)
        for i in range(1,n+1):
            pre[i] = pre[i-1]*nums[i-1]
            su[n-i] = su[n-i+1]*nums[n-i]
        answer = [0]*n
        for i in range(n):
            answer[i] = pre[i]*su[i+1]
        return answer
