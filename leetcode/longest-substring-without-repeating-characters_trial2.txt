class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        if len(s) == 0 or len(s) == 1:
            return len(s)
        start = 0 
        maxlen = 0 
        seen = {}

        for end in range(len(s)):
            if s[end] in seen and seen[s[end]]>= start:
                start = seen[s[end]] +1
            seen[s[end]] = end
            maxlen =  max(maxlen ,  end-start +1)

        return maxlen
