# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        prev, cur = dummy, head
        while cur and cur.next:
            nxt = cur.next
            cur.next = nxt.next
            nxt.next = cur
            prev.next = nxt
            prev = cur
            cur = cur.next
        return dummy.next
