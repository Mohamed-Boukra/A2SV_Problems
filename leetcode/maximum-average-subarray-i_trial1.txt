class Solution:
    def findMaxAverage(self, nums: List[int], k: int) -> float:
        n = len(nums)
        prefix = [0] * (n + 1)

        for i in range(n):
            prefix[i + 1] = prefix[i] + nums[i]

        maxSum = prefix[k]

        for i in range(n - k + 1):
            j = i + k - 1
            currSum = prefix[j + 1] - prefix[i]

            maxSum = max(maxSum, currSum)

        return maxSum /k
