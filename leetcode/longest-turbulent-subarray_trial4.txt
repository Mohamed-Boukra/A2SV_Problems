class Solution:
    def maxTurbulenceSize(self, arr: List[int]) -> int:
        result = 1
        l = 0
        for i in range(1,len(arr)):
            if arr[i-1] == arr[i]:
                l = i
            elif i>=2 and (arr[i-2]<arr[i-1]) == (arr[i-1]<arr[i]):
                l =i-1
            result = max(result , i-l+1)
        return result
