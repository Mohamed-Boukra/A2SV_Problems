class Solution:
    def countGood(self, nums: List[int], k: int) -> int:
        freq = defaultdict(int)
        pairs = 0
        l = 0
        result = 0
        for r in range(len(nums)):
            pairs += freq[nums[r]]
            freq[nums[r]] +=1
            while pairs >= k:
                freq[nums[l]] -= 1
                pairs -= freq[nums[l]]
                l+=1
            result +=l
            
        return result
