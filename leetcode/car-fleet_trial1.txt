class Solution:
    def carFleet(self, target: int, position: List[int], speed: List[int]) -> int:
        ps = list(zip(position , speed))
        ps.sort(reverse = True)
        stack = []
        for x , y in ps:
            t = (target - x)/y
            if not stack:
                stack.append(t)
            elif stack[-1] < t:
                stack.append(t)
        return len(stack)
