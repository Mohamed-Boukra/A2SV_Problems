# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        curr = head
        l = 1
        while curr.next:
            curr = curr.next
            l+=1
        curr.next = head
        k = k%l
        if k ==0:
            curr.next = None
            return head
        for _ in range(1,l - k):
            head = head.next
        result = head.next
        head.next = None
        return result
