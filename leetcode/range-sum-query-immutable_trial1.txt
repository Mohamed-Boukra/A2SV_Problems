class NumArray:

    def __init__(self, nums: List[int]):
        self.pre = [0] * (len(nums) + 1)
        for i, x in enumerate(nums):
            self.pre[i+1] = self.pre[i] + x

    def sumRange(self, left: int, right: int) -> int:
        return self.pre[right+1] - self.pre[left]
