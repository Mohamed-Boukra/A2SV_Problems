class Solution:
    def totalFruit(self, fruits: List[int]) -> int:
        l = 0
        basket = defaultdict(int)
        result = 0
        for r in range(len(fruits)):
            basket[fruits[r]] += 1
            if len(basket) > 2:
                basket[fruits[l]] -=1
                if basket[fruits[l]] == 0:
                    del basket[fruits[l]]
                l+=1
            result = max(result, r-l+1)
        return result
