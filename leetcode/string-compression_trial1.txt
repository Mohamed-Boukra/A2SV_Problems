class Solution:
    def compress(self, chars: List[str]) -> int:
        w = i = 0
        while i < len(chars):
            ch, count = chars[i], 0
            while i < len(chars) and chars[i] == ch:
                i += 1
                count += 1
            chars[w] = ch
            w += 1
            if count > 1:
                for c in str(count):
                    chars[w] = c
                    w += 1
        return w
